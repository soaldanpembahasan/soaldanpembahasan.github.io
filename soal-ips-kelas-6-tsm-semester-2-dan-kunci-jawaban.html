total nilai tsm = 60a

[tex]rata-rata = \frac{total nilai}{banyak siswa} \\\\69,5 = \frac{3040 +60a}{38 +a}[/tex]

69,5(38 + a) = 3040 + 60a

2641 + 69,5a = 3040 + 60a

69,5a - 60a = 3040 - 2641

9,5a = 399
