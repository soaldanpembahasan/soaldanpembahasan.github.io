[tex]t = \frac{t}{n} \\ t = \frac{2s}{4} \\ t = 0.5s[/tex]

[tex]f = \frac{n}{t} \\ f = \frac{4}{2s} \\ f = 2 \: hz[/tex]

[tex]w = \frac{2\pi}{t} \\ w = \frac{2\pi}{0.5} \\ w = 4\pi \frac{rad}{s} [/tex]

[tex]v = w \times r \\ v = 4\pi \frac{rad}{s} \times 14cm \\ v = 56\pi \frac{cm}{s} \\ v = 176 \frac{cm}{s} [/tex]

[tex]s = v \times 3t \\ s = 176 \frac{cm}{s} \times 3(0.5s) \\ s = 264cm[/tex]
